258  #version 440 core
259 
260  layout(location = 0) out vec4 colour;
261  in vec3 normal;
262  in vec3 fragmentPos;
263  in vec2 texCoord;
264  uniform vec3 u_lightPos;

266  uniform vec3 u_lightColour;
267  uniform sampler2D u_texData;
268  void main()

270  float ambientStrength = 0.4;

272  vec3 norm = normalize(normal);
273  vec3 lightDir = normalize(u_lightPos - fragmentPos);
274  float diff = max(dot(norm, lightDir), 0.0);

280  vec3 specular = specularStrength * spec * u_lightColour;
